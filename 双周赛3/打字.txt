利用哈希表的思想，就可以不用每一次都判断了。

#include<bits/stdc++.h>
int a[1001];
int b[9]; 
int main()
{
    a['A']=1;
    a['Q']=1;
    a['Z']=1;
    a['1']=1;
    a[27]=1;//ESC的ASCII值
	a['2']=a['W']=a['S']=a['X']=2;
	a['3']=a['E']=a['D']=a['C']=3;
	a['4']=a['R']=a['F']=a['V']=4;
	a['5']=a['T']=a['G']=a['B']=4;
	a['6']=a['Y']=a['H']=a['N']=5;
	a['7']=a['U']=a['J']=a['M']=5;
	a['8']=a['I']=a['K']=a[',']=6;
	a['9']=a['O']=a['L']=a['.']=7;
	a['0']=a['P']=a[';'] =a['/']=a['-']=a['=']=a['[']=a[']']=a[39]=8;//39是单引号的ASCII 
	char c;
	while(cin>>c)
	{
		b[a['c']]++;
	}
}

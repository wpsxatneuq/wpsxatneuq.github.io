看到题目先试一下枚举，枚举二进制的每一位，假设是错的，然后转化为10进制，再将得到的数转化为3进制，与所给的数进行判断，如果只有一位不一样，就找到了
code：
#include<bits/stdc++.h>
using namespace std;
string a,b,s;
long long ans,temp,x,t,sum;
int main(){
    cin>>a>>b;
    for(int k=0;k<a.size();k++){
        if(a[k]=='1' && k==0) continue;
        if(a[k]=='0') a[k]='1';
        else a[k]='0';
        ans=0;
        for(int i=0;i<a.size();i++) ans+=int(a[i]-'0')*pow(2,a.size()-i-1);
        temp=ans;
        x=0;
        while(temp>0){
            temp/=3;
            x++;
        }
        temp=ans;
        t=x-1;
        while(temp>1){
            s[t]=temp%3+'0';
            temp/=3;
            t--;
        }
        sum=0;
        if(x==b.size()){
            for(int i=0;i<b.size();i++){
                if(b[i]!=s[b.size()-i-1]) sum++;
            }
        }
        if(sum==1){
            cout<<ans;
            return 0;
        }
        if(a[k]=='0') a[k]='1';
        else a[k]='0';
    }
    return 0;
}
